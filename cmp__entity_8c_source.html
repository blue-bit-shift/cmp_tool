54 /**
55  * @brief calculate the size of the compression entity header based on a
56  * compression data product type
57  *
58  * @param data_type compression data product type
59  * @param raw_mode_flag set this flag if the raw compression mode (CMP_MODE_RAW) is used
60  *
61  * @returns size of the compression entity header in bytes, 0 on unknown data
62  * type
63  */
64 
65 uint32_t cmp_ent_cal_hdr_size(enum cmp_data_type data_type, int raw_mode_flag)
66 {
67  uint32_t size = 0;
68 
69  if (raw_mode_flag) {
70  if (!cmp_data_type_is_invalid(data_type))
71  /* for raw data we do not need a specific header */
72  size = GENERIC_HEADER_SIZE;
73  } else {
74  switch (data_type) {
75  case DATA_TYPE_IMAGETTE:
76  case DATA_TYPE_SAT_IMAGETTE:
77  case DATA_TYPE_F_CAM_IMAGETTE:
78  size = IMAGETTE_HEADER_SIZE;
79  break;
80  case DATA_TYPE_IMAGETTE_ADAPTIVE:
81  case DATA_TYPE_SAT_IMAGETTE_ADAPTIVE:
82  case DATA_TYPE_F_CAM_IMAGETTE_ADAPTIVE:
83  size = IMAGETTE_ADAPTIVE_HEADER_SIZE;
84  break;
85  case DATA_TYPE_OFFSET:
86  case DATA_TYPE_BACKGROUND:
87  case DATA_TYPE_SMEARING:
88  case DATA_TYPE_S_FX:
89  case DATA_TYPE_S_FX_EFX:
90  case DATA_TYPE_S_FX_NCOB:
91  case DATA_TYPE_S_FX_EFX_NCOB_ECOB:
92  case DATA_TYPE_L_FX:
93  case DATA_TYPE_L_FX_EFX:
94  case DATA_TYPE_L_FX_NCOB:
95  case DATA_TYPE_L_FX_EFX_NCOB_ECOB:
96  case DATA_TYPE_F_FX:
97  case DATA_TYPE_F_FX_EFX:
98  case DATA_TYPE_F_FX_NCOB:
99  case DATA_TYPE_F_FX_EFX_NCOB_ECOB:
100  case DATA_TYPE_F_CAM_OFFSET:
101  case DATA_TYPE_F_CAM_BACKGROUND:
102  size = NON_IMAGETTE_HEADER_SIZE;
103  break;
104  case DATA_TYPE_UNKNOWN:
105  size = 0;
106  break;
107  }
108  }
109  return size;
110 }
