30 /**
31  * @brief create an RDCU compression configuration
32  *
33  * @param data_type compression data product type
34  * @param cmp_mode compression mode
35  * @param model_value model weighting parameter (only needed for model compression mode)
36  * @param lossy_par lossy rounding parameter (use CMP_LOSSLESS for lossless compression)
37  *
38  * @returns a compression configuration containing the chosen parameters;
39  * on error the data_type record is set to DATA_TYPE_UNKNOWN
40  */
41 
42 struct cmp_cfg rdcu_cfg_create(enum cmp_data_type data_type, enum cmp_mode cmp_mode,
43  uint32_t model_value, uint32_t lossy_par)
44 {
45  struct cmp_cfg cfg;
46 
47  memset(&cfg, 0, sizeof(cfg));
48 
49  cfg.data_type = data_type;
50  cfg.cmp_mode = cmp_mode;
51  cfg.model_value = model_value;
52  cfg.round = lossy_par;
53  cfg.max_used_bits = &MAX_USED_BITS_SAFE;
54 
55  if (cmp_cfg_gen_par_is_invalid(&cfg, RDCU_CHECK))
56  cfg.data_type = DATA_TYPE_UNKNOWN;
57 
58  return cfg;
59 }

62 /**
63  * @brief check if a buffer is in outside the RDCU SRAM
64  *
65  * @param addr start address of the buffer
66  * @param size length of the buffer in bytes
67  *
68  * @returns 0 if buffer in inside the RDCU SRAM, 1 when the buffer is outside
69  */
70 
71 static int outside_sram_range(uint32_t addr, uint32_t size)
72 {
73  if (addr + size > RDCU_SRAM_START + RDCU_SRAM_SIZE)
74  return 1;
75 
76  if (addr > RDCU_SRAM_END)
77  return 1;
78 
79  if (size > RDCU_SRAM_SIZE)
80  return 1;
81 
82  return 0;
83 }

86 /**
87  * @brief check if two buffers are overlapping
88  * @note implement according to https://stackoverflow.com/a/325964
89  *
90  * @param start_a start address of the 1st buffer
91  * @param end_a end address of the 1st buffer
92  * @param start_b start address of the 2nd buffer
93  * @param end_b end address of the 2nd buffer
94  *
95  * @returns 0 if buffers are not overlapping, otherwise buffer are
96  * overlapping
97  */
98 
99 static int buffers_overlap(uint32_t start_a, uint32_t end_a, uint32_t start_b,
100  uint32_t end_b)
101 {
102  if (start_a < end_b && end_a > start_b)
103  return 1;
104  else
105  return 0;
106 }

109 /**
110  * @brief check if RDCU buffer settings are invalid
111  *
112  * @param cfg a pointer to a compression configuration
113  *
114  * @returns 0 if buffers configuration is valid, otherwise the configuration is
115  * invalid
116  */
117 
118 static int rdcu_cfg_buffers_is_invalid(const struct cmp_cfg *cfg)
119 {
120  int cfg_invalid = 0;
121 
122  if (cfg->cmp_mode == CMP_MODE_RAW) {
123  if (cfg->buffer_length < cfg->samples) {
124  debug_print("rdcu_buffer_length is smaller than the samples parameter. There is not enough space to copy the data in RAW mode.\n");
125  cfg_invalid++;
126  }
127  }
128 
129  if (cfg->rdcu_data_adr & 0x3) {
130  debug_print("Error: The RDCU data to compress start address is not 4-Byte aligned.\n");
131  cfg_invalid++;
132  }
133 
134  if (cfg->rdcu_buffer_adr & 0x3) {
135  debug_print("Error: The RDCU compressed data start address is not 4-Byte aligned.\n");
136  cfg_invalid++;
137  }
138 
139  if (outside_sram_range(cfg->rdcu_data_adr, cfg->samples * IMA_SAM2BYT)) {
140  debug_print("Error: The RDCU data to compress buffer is outside the RDCU SRAM address space.\n");
141  cfg_invalid++;
142  }
143 
144  if (outside_sram_range(cfg->rdcu_buffer_adr, cfg->buffer_length * IMA_SAM2BYT)) {
145  debug_print("Error: The RDCU compressed data buffer is outside the RDCU SRAM address space.\n");
146  cfg_invalid++;
147  }
148 
149  if (buffers_overlap(cfg->rdcu_data_adr,
150  cfg->rdcu_data_adr + cfg->samples * IMA_SAM2BYT,
151  cfg->rdcu_buffer_adr,
152  cfg->rdcu_buffer_adr + cfg->buffer_length * IMA_SAM2BYT)) {
153  debug_print("Error: The RDCU data to compress buffer and the RDCU compressed data buffer are overlapping.\n");
154  cfg_invalid++;
155  }
156 
157  if (model_mode_is_used(cfg->cmp_mode)) {
158  if (cfg->model_buf && cfg->model_buf == cfg->input_buf) {
159  debug_print("Error: The model buffer (model_buf) and the data to be compressed (input_buf) are equal.\n");
160  cfg_invalid++;
161  }
162 
163  if (cfg->rdcu_model_adr & 0x3) {
164  debug_print("Error: The RDCU model start address is not 4-Byte aligned.\n");
165  cfg_invalid++;
166  }
167 
168  if (outside_sram_range(cfg->rdcu_model_adr, cfg->samples * IMA_SAM2BYT)) {
169  debug_print("Error: The RDCU model buffer is outside the RDCU SRAM address space.\n");
170  cfg_invalid++;
171  }
172 
173  if (buffers_overlap(
174  cfg->rdcu_model_adr,
175  cfg->rdcu_model_adr + cfg->samples * IMA_SAM2BYT,
176  cfg->rdcu_data_adr,
177  cfg->rdcu_data_adr + cfg->samples * IMA_SAM2BYT)) {
178  debug_print("Error: The model buffer and the data to compress buffer are overlapping.\n");
179  cfg_invalid++;
180  }
181 
182  if (buffers_overlap(
183  cfg->rdcu_model_adr,
184  cfg->rdcu_model_adr + cfg->samples * IMA_SAM2BYT,
185  cfg->rdcu_buffer_adr,
186  cfg->rdcu_buffer_adr + cfg->buffer_length * IMA_SAM2BYT)
187  ) {
188  debug_print("Error: The model buffer and the compressed data buffer are overlapping.\n");
189  cfg_invalid++;
190  }
191 
192  if (cfg->rdcu_model_adr != cfg->rdcu_new_model_adr) {
193  if (cfg->rdcu_new_model_adr & 0x3) {
194  debug_print("Error: The RDCU updated model start address (rdcu_new_model_adr) is not 4-Byte aligned.\n");
195  cfg_invalid++;
196  }
197 
198  if (outside_sram_range(cfg->rdcu_new_model_adr,
199  cfg->samples * IMA_SAM2BYT)) {
200  debug_print("Error: The RDCU updated model buffer is outside the RDCU SRAM address space.\n");
201  cfg_invalid++;
202  }
203 
204  if (buffers_overlap(
205  cfg->rdcu_new_model_adr,
206  cfg->rdcu_new_model_adr + cfg->samples * IMA_SAM2BYT,
207  cfg->rdcu_data_adr,
208  cfg->rdcu_data_adr + cfg->samples * IMA_SAM2BYT)
209  ) {
210  debug_print("Error: The updated model buffer and the data to compress buffer are overlapping.\n");
211  cfg_invalid++;
212  }
213 
214  if (buffers_overlap(
215  cfg->rdcu_new_model_adr,
216  cfg->rdcu_new_model_adr + cfg->samples * IMA_SAM2BYT,
217  cfg->rdcu_buffer_adr,
218  cfg->rdcu_buffer_adr + cfg->buffer_length * IMA_SAM2BYT)
219  ) {
220  debug_print("Error: The updated model buffer and the compressed data buffer are overlapping.\n");
221  cfg_invalid++;
222  }
223  if (buffers_overlap(
224  cfg->rdcu_new_model_adr,
225  cfg->rdcu_new_model_adr + cfg->samples * IMA_SAM2BYT,
226  cfg->rdcu_model_adr,
227  cfg->rdcu_model_adr + cfg->samples * IMA_SAM2BYT)
228  ) {
229  debug_print("Error: The updated model buffer and the model buffer are overlapping.\n");
230  cfg_invalid++;
231  }
232  }
233  }
234 
235 #ifdef SKIP_CMP_PAR_CHECK
236  return 0;
237 #endif
238  return cfg_invalid;
239 }

242 /**
243  * @brief setup of the different data buffers for an RDCU compression
244  *
245  * @param cfg pointer to a compression configuration (created
246  * with the rdcu_cfg_create() function)
247  * @param data_to_compress pointer to the data to be compressed (if NULL no
248  * data transfer to the RDCU)
249  * @param data_samples length of the data to be compressed (plus the
250  * collection header) measured in 16-bit data samples
251  * @param model_of_data pointer to the model data buffer (only needed for
252  * model compression mode, if NULL no model data is
253  * transferred to the RDCU)
254  * @param rdcu_data_adr RDCU SRAM data to compress start address
255  * @param rdcu_model_adr RDCU SRAM model start address (only needed for
256  * model compression mode)
257  * @param rdcu_new_model_adr RDCU SRAM new/updated model start address (can be
258  * the same as rdcu_model_adr for in-place model update)
259  * @param rdcu_buffer_adr RDCU SRAM compressed data start address
260  * @param rdcu_buffer_lenght length of the RDCU compressed data SRAM buffer
261  * measured in 16-bit units (same as data_samples)
262  *
263  * @returns 0 if parameters are valid, non-zero if parameters are invalid
264  */
265 
266 int rdcu_cfg_buffers(struct cmp_cfg *cfg, uint16_t *data_to_compress,
267  uint32_t data_samples, uint16_t *model_of_data,
268  uint32_t rdcu_data_adr, uint32_t rdcu_model_adr,
269  uint32_t rdcu_new_model_adr, uint32_t rdcu_buffer_adr,
270  uint32_t rdcu_buffer_lenght)
271 {
272  if (!cfg) {
273  debug_print("Error: pointer to the compression configuration structure is NULL.\n");
274  return -1;
275  }
276 
277  cfg->input_buf = data_to_compress;
278  cfg->samples = data_samples;
279  cfg->model_buf = model_of_data;
280  cfg->rdcu_data_adr = rdcu_data_adr;
281  cfg->rdcu_model_adr = rdcu_model_adr;
282  cfg->rdcu_new_model_adr = rdcu_new_model_adr;
283  cfg->rdcu_buffer_adr = rdcu_buffer_adr;
284  cfg->buffer_length = rdcu_buffer_lenght;
285 
286  return rdcu_cfg_buffers_is_invalid(cfg);
287 }

359 /**
360  * @brief check if the compressor configuration is invalid for an RDCU compression,
361  * see the user manual for more information (PLATO-UVIE-PL-UM-0001).
362  *
363  * @param cfg pointer to a compression configuration contains all parameters
364  * required for compression
365  *
366  * @returns 0 if parameters are valid, non-zero if parameters are invalid
367  */
368 
369 int rdcu_cmp_cfg_is_invalid(const struct cmp_cfg *cfg)
370 {
371  int cfg_invalid = 0;
372 
373  if (!cfg) {
374  debug_print("Error: pointer to the compression configuration structure is NULL.\n");
375  return -1;
376  }
377 
378  if (!cfg->input_buf)
379  debug_print("Warning: The data to compress buffer is set to NULL. No data will be transferred to the rdcu_data_adr in the RDCU SRAM.\n");
380 
381  if (model_mode_is_used(cfg->cmp_mode)) {
382  if (!cfg->model_buf)
383  debug_print("Warning: The model buffer is set to NULL. No model data will be transferred to the rdcu_model_adr in the RDCU SRAM.\n");
384  }
385 
386  if (cfg->samples == 0)
387  debug_print("Warning: The samples parameter is set to 0. No data will be compressed.\n");
388 
389  if (cfg->icu_new_model_buf)
390  debug_print("Warning: ICU updated model buffer is set. This buffer is not used for an RDCU compression.\n");
391 
392  if (cfg->icu_output_buf)
393  debug_print("Warning: ICU compressed data buffer is set. This buffer is not used for an RDCU compression.\n");
394 
395  if (cfg->buffer_length == 0) {
396  debug_print("Error: The buffer_length is set to 0. There is no place to store the compressed data.\n");
397  cfg_invalid++;
398  }
399 
400  cfg_invalid += cmp_cfg_gen_par_is_invalid(cfg, RDCU_CHECK);
401  cfg_invalid += rdcu_cfg_buffers_is_invalid(cfg);
402  cfg_invalid += cmp_cfg_imagette_is_invalid(cfg, RDCU_CHECK);
403 
404  return cfg_invalid;
405 }
