74  *
75  * @param dec a pointer to a bit_decoder context
76  * @param unused_1 this parameter is not used
77  * @param unused_2 this parameter is not used

83 static __inline uint32_t unary_decoder(struct bit_decoder *dec, uint32_t unused_1,
84  uint32_t unused_2)
85 {
86  uint32_t const decoded_cw = bit_peek_leading_ones(dec); /* decode unary coding */
87  uint32_t const cw_len = decoded_cw + 1; /* Number of 1's + following 0 */
88 
89  UNUSED(unused_1); /* we don't need this parameter */
90  UNUSED(unused_2); /* and this parameter */
91 
92  bit_consume_bits(dec, cw_len);
93 
94  return decoded_cw;

98 /**
99  * @brief decode the next Rice code word in the bitstream
100  *
101  * @param dec a pointer to a bit_decoder context
102  * @param m Golomb parameter, must be the same used for encoding
103  * @param log2_m Rice parameter, is ilog_2(m), must be larger than 0
104  * @note the Golomb parameter (m) must be a power of 2
105  * @warning the Rice parameter (log2_m) must be greater than 0! If you want to
106  * use a Rice parameter equal to 0, use the unary_decoder instead.
107  *
108  * @returns the decoded value
109  */
110 
111 static uint32_t rice_decoder(struct bit_decoder *dec, uint32_t m, uint32_t log2_m)
112 {
113  uint32_t q; /* quotient */
114  uint32_t r; /* remainder */
115 
116  assert(log2_m > 0 && log2_m < 32);
117 
118  q = unary_decoder(dec, m, log2_m); /* decode quotient unary code part */
119  r = bit_read_bits32(dec, log2_m); /* get remainder */
120 
121  return (q << log2_m) + r; /* calculate decoded value (q*m+r) */
122 }

125 /**
126  * @brief decode the next Golomb code word in the bitstream
127  *
128  * @param dec a pointer to a bit_decoder context
129  * @param m Golomb parameter (have to be bigger than 0)
130  * @param log2_m is ilog_2(m) calculate outside function for better
131  * performance
132  *
133  * @returns the decoded value
134  */
135 
136 static uint32_t golomb_decoder(struct bit_decoder *dec, uint32_t m, uint32_t log2_m)
137 {
138  uint32_t q; /* quotient */
139  uint32_t r1; /* remainder case 1 */
140  uint32_t r2; /* remainder case 2 */
141  uint32_t r; /* remainder */
142  uint32_t cutoff; /* cutoff between group 1 and 2 */
143 
144  assert(m > 0);
145  assert(log2_m == ilog_2(m));
146 
147  /* decode quotient unary code part */
148  q = unary_decoder(dec, m, log2_m);
149 
150  /* get the remainder code for both cases */
151  r2 = (uint32_t)bit_peek_bits(dec, log2_m+1);
152  r1 = r2 >> 1;
153 
154  /* calculate cutoff between case 1 and 2 */
155  cutoff = (0x2U << log2_m) - m; /* = 2^(log2_m+1)-m */
156 
157  if (r1 < cutoff) { /* remainder case 1: remainder length=log2_m */
158  bit_consume_bits(dec, log2_m);
159  r = r1;
160  } else { /* remainder case 2: remainder length = log2_m+1 */
161  bit_consume_bits(dec, log2_m+1);
162  r = r2 - cutoff;
163  }
164 
165  return q*m + r;
166 }

203 static int decode_zero(const struct decoder_setup *setup, uint32_t *decoded_value)
204 {
205  /* Decode the next value in the bitstream with the Golomb/Rice/unary decoder */
206  *decoded_value = setup->decode_cw_f(setup->dec, setup->encoder_par1, setup->encoder_par2);
207 
208  if (*decoded_value != 0) { /* no escape symbol detected */
209  if (*decoded_value >= setup->outlier_par) {
210  debug_print("Error: Data consistency check failed. Non-outlier decoded value greater or equal than the outlier parameter. %s", please_check_str);
211  return CORRUPTION_DETECTED;
212  }
213  *decoded_value -= 1;
214  } else {
215  /* the zero escape symbol mechanism was used; read unencoded value */
216  bit_refill(setup->dec);
217  *decoded_value = bit_read_bits32_sub_1(setup->dec, setup->max_data_bits);
218 
219  if (*decoded_value < setup->outlier_par - 1) { /* -1 because we subtract -1 from the *decoded_value */
220  if (bit_refill(setup->dec) != BIT_OVERFLOW)
221  debug_print("Error: Data consistency check failed. Outlier small than the outlier parameter. %s", please_check_str);
222  return CORRUPTION_DETECTED;
223  }
224  }
225  return bit_refill(setup->dec) == BIT_OVERFLOW;
226 }

230  * @brief decode the next code word with the multi escape mechanism from the bitstream
231  *
232  * @param setup pointer to the decoder setup
233  * @param decoded_value points to the location where the decoded value is stored
234  *
235  * @returns 0 on success; otherwise error

237 
238 static int decode_multi(const struct decoder_setup *setup, uint32_t *decoded_value)
239 {
240  /* Decode the next value in the bitstream with the Golomb/Rice/unary decoder */
241  *decoded_value = setup->decode_cw_f(setup->dec, setup->encoder_par1, setup->encoder_par2);
242 
243  if (*decoded_value >= setup->outlier_par) { /* escape symbol mechanism detected */
244  uint32_t const unencoded_len = (*decoded_value - setup->outlier_par + 1) << 1;
245 
246  if (unencoded_len > ((setup->max_data_bits+1) & -2U)) { /* round up max_data_bits to the nearest multiple of 2 */
247  debug_print("Error: Data consistency check failed. Multi escape symbol higher than expected. %s", please_check_str);
248  return CORRUPTION_DETECTED;
249  }
250 
251  /* read unencoded value */
252  bit_refill(setup->dec);
253  *decoded_value = bit_read_bits32(setup->dec, unencoded_len);
254 
255  if (*decoded_value >> (unencoded_len-2) == 0) { /* check if at least one bit of the two highest is set. */
256  if (unencoded_len > 2) { /* Exception: if we code outlier_par as outlier, no set bit is expected */
257  if (bit_refill(setup->dec) != BIT_OVERFLOW)
258  debug_print("Error: Data consistency check failed. Unencoded data after multi escape symbol to small. %s", please_check_str);
259  return CORRUPTION_DETECTED;
260  }
261  }
262 
263  *decoded_value += setup->outlier_par;
264 
265  if ((*decoded_value & BIT_MASK[setup->max_data_bits]) < setup->outlier_par) { /* check for overflow in addition */
266  if (bit_refill(setup->dec) != BIT_OVERFLOW)
267  debug_print("Error: Data consistency check failed. Outlier small than the outlier parameter. %s", please_check_str);
268  return CORRUPTION_DETECTED;
269  }
270  }
271  return bit_refill(setup->dec) == BIT_OVERFLOW;
272 }

295  * @note this is the reverse function of map_to_pos()
296  *
297  * @param value_to_unmap unsigned value to remap
298  *

301 
302 static __inline uint32_t re_map_to_pos(uint32_t value_to_unmap)
303 {
304  if (value_to_unmap & 0x1) { /* if uneven */
305  /* uint64_t to prevent overflow if value_to_unmap == 0xFFFFFFFF */

309  } else {
310  return value_to_unmap / 2;

315 /**
316  * @brief decompress the next code word in the bitstream and decorate it with
317  * the model
318  *
319  * @param setup pointer to the decoder setup
320  * @param decoded_value points to the location where the decoded value is stored
321  * @param model model of the decoded_value (0 if not used)
322  *
323  * @returns 0 on success; otherwise error
324  */
325 
326 static int decode_value(const struct decoder_setup *setup, uint32_t *decoded_value,
327  uint32_t model)
328 {
329  /* decode the next value from the bitstream */
330  int err = setup->decode_method_f(setup, decoded_value);
331 
332  if (setup->decode_method_f == decode_none)
333  /* we are done here in stuff mode */
334  return err;
335 
336  /* map the unsigned decode value back to a signed value */
337  *decoded_value = re_map_to_pos(*decoded_value);
338 
339  /* decorate data the data with the model */
340  *decoded_value += round_fwd(model, setup->lossy_par);
341 
342  /* we mask only the used bits in case there is an overflow when adding the model */
343  *decoded_value &= BIT_MASK[setup->max_data_bits];
344 
345  /* inverse step of the lossy compression */
346  *decoded_value = round_inv(*decoded_value, setup->lossy_par);
347 
348  return err;
349 }

353  * @brief configure a decoder setup structure to have a setup to decode a vale
354  *
355  * @param setup pointer to the decoder setup
356  * @param dec pointer to a bit_decoder context
357  * @param cmp_mode compression mode
358  * @param cmp_par compression parameter

361  * @param max_data_bits how many bits are needed to represent the highest possible value

363 
364 static void configure_decoder_setup(struct decoder_setup *setup, struct bit_decoder *dec,
365  enum cmp_mode cmp_mode, uint32_t cmp_par,
366  uint32_t spillover, uint32_t lossy_par,
367  uint32_t max_data_bits)
368 {
369  assert(setup != NULL);
370  assert(dec != NULL);
371  assert(cmp_par != 0);
372  assert(max_data_bits > 0 && max_data_bits <= 32);
373 
374  if (multi_escape_mech_is_used(cmp_mode))
375  setup->decode_method_f = &decode_multi;
376  else if (zero_escape_mech_is_used(cmp_mode))
377  setup->decode_method_f = &decode_zero;
378  else if (cmp_mode == CMP_MODE_STUFF)
379  setup->decode_method_f = &decode_none;
380  else {
381  debug_print("Error: Compression mode not supported.\n");
382  assert(0);
383  }
384  setup->decode_cw_f = select_decoder(cmp_par);
385  setup->dec = dec;
386  setup->encoder_par1 = cmp_par; /* encoding parameter 1 */
387  setup->encoder_par2 = ilog_2(cmp_par); /* encoding parameter 2 */
388  setup->outlier_par = spillover; /* outlier parameter */
389  setup->lossy_par = lossy_par; /* lossy compression parameter */
390  setup->max_data_bits = max_data_bits; /* how many bits are needed to represent the highest possible value */
391 }
