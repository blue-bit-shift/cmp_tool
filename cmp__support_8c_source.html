338 /**
339  * @brief get the maximum valid spill threshold value for a non-imagette compression
340  * in diff or model mode
341  *
342  * @param cmp_par compression parameter
343  *
344  * @returns the highest still valid spill threshold value for diff or model
345  * mode compression; 0 if the cmp_par is not valid
346  */
347 
348 uint32_t cmp_icu_max_spill(unsigned int cmp_par)
349 {
350  /* the ICU compressor can generate code words with a length of maximal 32 bits. */
351  unsigned int const max_cw_bits = 32;
352  unsigned int const cutoff = (0x2U << (ilog_2(cmp_par) & 0x1FU)) - cmp_par;
353  unsigned int const max_n_sym_offset = max_cw_bits/2 - 1;
354 
355  if (!cmp_par || cmp_par > MAX_NON_IMA_GOLOMB_PAR)
356  return 0;
357 
358  return (max_cw_bits-1-ilog_2(cmp_par))*cmp_par + cutoff
359  - max_n_sym_offset - 1;
360 }

608 /**
609  * @brief check if the combination of the different compression parameters is invalid
610  *
611  * @param cmp_par compression parameter
612  * @param spill spillover threshold parameter
613  * @param cmp_mode compression mode
614  * @param data_type compression data type
615  * @param par_name string describing the use of the compression par. for
616  * debug messages (can be NULL)
617  *
618  * @returns 0 if the parameter combination is valid, otherwise the combination is invalid
619  */
620 
621 static int cmp_pars_are_invalid(uint32_t cmp_par, uint32_t spill, enum cmp_mode cmp_mode,
622  enum cmp_data_type data_type, const char *par_name)
623 {
624  int cfg_invalid = 0;
625  uint32_t min_golomb_par;
626  uint32_t max_golomb_par;
627  uint32_t min_spill;
628  uint32_t max_spill;
629 
630  /* The maximum compression parameter for imagette data are smaller to
631  * fit into the imagette compression entity header */
632  if (cmp_imagette_data_type_is_used(data_type)) {
633  min_golomb_par = MIN_IMA_GOLOMB_PAR;
634  max_golomb_par = MAX_IMA_GOLOMB_PAR;
635  min_spill = MIN_IMA_SPILL;
636  max_spill = cmp_ima_max_spill(cmp_par);
637  } else {
638  min_golomb_par = MIN_NON_IMA_GOLOMB_PAR;
639  max_golomb_par = MAX_NON_IMA_GOLOMB_PAR;
640  min_spill = MIN_NON_IMA_SPILL;
641  max_spill = cmp_icu_max_spill(cmp_par);
642  }
643 
644 
645  switch (cmp_mode) {
646  case CMP_MODE_RAW:
647  /* no checks needed */
648  break;
649  case CMP_MODE_DIFF_ZERO:
650  case CMP_MODE_DIFF_MULTI:
651  case CMP_MODE_MODEL_ZERO:
652  case CMP_MODE_MODEL_MULTI:
653  if (cmp_par < min_golomb_par || cmp_par > max_golomb_par) {
654  debug_print("Error: The selected %s compression parameter: %" PRIu32 " is not supported in the selected compression mode. The compression parameter has to be between [%" PRIu32 ", %" PRIu32 "] in this mode.\n",
655  par_name, cmp_par, min_golomb_par, max_golomb_par);
656  cfg_invalid++;
657  }
658  if (spill < min_spill) {
659  debug_print("Error: The selected %s spillover threshold value: %" PRIu32 " is too small. The smallest possible spillover value is: %" PRIu32 ".\n",
660  par_name, spill, min_spill);
661  cfg_invalid++;
662  }
663  if (spill > max_spill) {
664  debug_print("Error: The selected %s spillover threshold value: %" PRIu32 " is too large for the selected %s compression parameter: %" PRIu32 ". The largest possible spillover value in the selected compression mode is: %" PRIu32 ".\n",
665  par_name, spill, par_name, cmp_par, max_spill);
666  cfg_invalid++;
