47 /**
48  * @brief source data header structure for multi entry packet
49  * @note a scientific package contains a multi-entry header followed by multiple

55 struct multi_entry_hdr {

57  uint16_t timestamp_fine;
58  uint16_t configuration_id;
59  uint16_t collection_id;
60  uint16_t collection_length;
