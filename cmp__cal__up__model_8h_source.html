84 /* fast calculation for data size smaller that uint32_t */
85 static inline uint16_t cmp_up_model16(uint32_t data, uint32_t model,
86  unsigned int model_value, unsigned int round)
87 {
88  /* round and round back input because for decompression the accurate
89  * data values are not available
90  */
91  uint32_t const weighted_data = round_inv(round_fwd(data, round), round)
92  * (MAX_MODEL_VALUE - model_value);
93  uint32_t const weighted_model = model * model_value;
94 
95  /* truncation is intended */
96  return (uint16_t)((weighted_model + weighted_data) / MAX_MODEL_VALUE);
97 }

100 /* slow calculation for uint32_t data size */
101 static inline uint32_t cmp_up_model32(uint32_t data, uint32_t model,
102  unsigned int model_value, unsigned int round)
103 {
104  /* round and round back input because for decompression the accurate
105  * data values are not available
106  * cast to uint64_t to prevent overflow in the multiplication
107  */
108  uint64_t const weighted_data = (uint64_t)round_inv(round_fwd(data, round), round)
109  * (MAX_MODEL_VALUE - model_value);
110  uint64_t const weighted_model = (uint64_t)model * model_value;
111 
112  /* truncation is intended */
113  return (uint32_t)((weighted_model + weighted_data) / MAX_MODEL_VALUE);
114 }
