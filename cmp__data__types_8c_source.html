107 /**
108  * @brief calculate the need bytes for the data
109  *
110  * @param samples number of data samples
111  * @param data_type compression data_type
112  *
113  * @note for non-imagette data program types the multi entry header size is added
114  *
115  * @returns the size in bytes to store the data sample; zero on failure
116  */
117 
118 uint32_t cmp_cal_size_of_data(uint32_t samples, enum cmp_data_type data_type)
119 {
120  size_t s = size_of_a_sample(data_type);
121  uint64_t x; /* use 64 bit to catch overflow */
122 
123  if (!s)
124  return 0;
125 
126  x = (uint64_t)s*samples;
127 
128  if (!rdcu_supported_data_type_is_used(data_type))
129  x += MULTI_ENTRY_HDR_SIZE;
130 
131  if (x > UINT_MAX) /* catch overflow */
132  return 0;
133 
134  return (unsigned int)x;
135 }

138 /**
139  * @brief calculates the number of samples for a given data size for the
140  * different compression modes
141  *
142  * @param size size of the data in bytes
143  * @param data_type compression data type
144  *
145  * @returns the number samples for the given compression mode; negative on error
146  */
147 
148 int32_t cmp_input_size_to_samples(uint32_t size, enum cmp_data_type data_type)
149 {
150  uint32_t samples_size = (uint32_t)size_of_a_sample(data_type);
151 
152  if (!samples_size)
153  return -1;
154 
155  if (!rdcu_supported_data_type_is_used(data_type)) {
156  if (size < MULTI_ENTRY_HDR_SIZE)
157  return -1;
158  size -= MULTI_ENTRY_HDR_SIZE;
159  }
160 
161  if (size % samples_size)
162  return -1;
163 
164  return (int)(size/samples_size);
165 }

372 /**
373  * @brief swap the endianness of uncompressed data form big endian to the cpu
374  * endianness (or the other way around) in place
375  *
376  * @param data pointer to a data sample
377  * @param data_size_byte size of the data in bytes
378  * @param data_type compression data type
379  *
380  * @returns 0 on success; non-zero on failure
381  */
382 
383 int cmp_input_big_to_cpu_endianness(void *data, uint32_t data_size_byte,
384  enum cmp_data_type data_type)
385 {
386  int32_t samples = cmp_input_size_to_samples(data_size_byte, data_type);
387 
388  if (!data) /* nothing to do */
389  return 0;
390 
391  if (samples < 0) {
392  debug_print("Error: Can not convert data size in samples.\n");
393  return -1;
394  }
395 
396  /* we do not convert the endianness of the multi entry header */
397  if (!rdcu_supported_data_type_is_used(data_type))
398  data = (uint8_t *)data + MULTI_ENTRY_HDR_SIZE;
399 
400  switch (data_type) {
401  case DATA_TYPE_IMAGETTE:
402  case DATA_TYPE_IMAGETTE_ADAPTIVE:
403  case DATA_TYPE_SAT_IMAGETTE:
404  case DATA_TYPE_SAT_IMAGETTE_ADAPTIVE:
405  case DATA_TYPE_F_CAM_IMAGETTE:
406  case DATA_TYPE_F_CAM_IMAGETTE_ADAPTIVE:
407  be_to_cpus_16(data, samples);
408  break;
409  case DATA_TYPE_OFFSET:
410  case DATA_TYPE_F_CAM_OFFSET:
411  be_to_cpus_offset(data, samples);
412  break;
413  case DATA_TYPE_BACKGROUND:
414  case DATA_TYPE_F_CAM_BACKGROUND:
415  be_to_cpus_background(data, samples);
416  break;
417  case DATA_TYPE_SMEARING:
418  be_to_cpus_smearing(data, samples);
419  break;
420  case DATA_TYPE_S_FX:
421  be_to_cpus_s_fx(data, samples);
422  break;
423  case DATA_TYPE_S_FX_EFX:
424  be_to_cpus_s_fx_efx(data, samples);
425  break;
426  case DATA_TYPE_S_FX_NCOB:
427  be_to_cpus_s_fx_ncob(data, samples);
428  break;
429  case DATA_TYPE_S_FX_EFX_NCOB_ECOB:
430  be_to_cpus_s_fx_efx_ncob_ecob(data, samples);
431  break;
432  case DATA_TYPE_L_FX:
433  be_to_cpus_l_fx(data, samples);
434  break;
435  case DATA_TYPE_L_FX_EFX:
436  be_to_cpus_l_fx_efx(data, samples);
437  break;
438  case DATA_TYPE_L_FX_NCOB:
439  be_to_cpus_l_fx_ncob(data, samples);
440  break;
441  case DATA_TYPE_L_FX_EFX_NCOB_ECOB:
442  be_to_cpus_l_fx_efx_ncob_ecob(data, samples);
443  break;
444  case DATA_TYPE_F_FX:
445  be_to_cpus_f_fx(data, samples);
446  break;
447  case DATA_TYPE_F_FX_EFX:
448  be_to_cpus_f_fx_efx(data, samples);
449  break;
450  case DATA_TYPE_F_FX_NCOB:
451  be_to_cpus_f_fx_ncob(data, samples);
452  break;
453  case DATA_TYPE_F_FX_EFX_NCOB_ECOB:
454  be_to_cpus_f_fx_efx_ncob_ecob(data, samples);
455  break;
456  /* LCOV_EXCL_START */
457  case DATA_TYPE_UNKNOWN:
458  default:
459  debug_print("Error: Can not swap endianness for this compression data type.\n");
460  return -1;
461  /* LCOV_EXCL_STOP */
462  }
463 
464  return 0;
