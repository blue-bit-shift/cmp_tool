115 /**
116  * While the size of a SpW packet is in principl not limited, the size of the
117  * header cannot be more than 255 bytes given the 8-bit width of the transfer
118  * descriptor's HEADERLEN field in the GRSPW2 core, so we'll use that as our
119  * limit.
120  *
121  * The reply address path can be at most 12 bytes, as the reply address length
122  * field in the RMAP instruction field is only 2 bits wide and counts the
123  * number of 32 bit words needed to hold the return path.
124  *
125  * The maximum data length is 2^24-1 bits
126  *
127  * All other fields in the header (not counting the header CRC) amount to
128  * 27 bytes.
129  *
130  * @see GR712RC-UM v2.7 p112 and ECSS‐E‐ST‐50‐52C e.g. 5.3.1.1
131  */
132 
133 #define RMAP_MAX_PATH_LEN 15
134 #define RMAP_MAX_REPLY_ADDR_LEN 3
135 #define RMAP_MAX_REPLY_PATH_LEN 12
136 #define RMAP_MAX_DATA_LEN 0xFFFFFFUL
