112 static int rdcu_set_compression_register(const struct cmp_cfg *cfg)
113 {
114  if (rdcu_cmp_cfg_is_invalid(cfg))
115  return -1;
116 #if 1
117  /*
118  * There is a bug in the RDCU HW data compressor, when a non-raw mode
119  * compression is performed after a raw mode compression, the compressor
120  * gets stuck due to a deadlock condition. Performing a compression
121  * interrupt after a raw mode compression work around this bug.
122  */
123  if (rdcu_sync_used_param1())
124  return -1;
125  rdcu_syncing();
126  if (rdcu_get_compression_mode() == CMP_MODE_RAW)
127  rdcu_interrupt_compression();
128 #endif
129 
130 
131  /* first, set compression parameters in local mirror registers */
132  if (rdcu_set_compression_mode(cfg->cmp_mode))
133  return -1;
134  if (rdcu_set_golomb_param(cfg->golomb_par))
135  return -1;
136  if (rdcu_set_spillover_threshold(cfg->spill))
137  return -1;
138  if (rdcu_set_weighting_param(cfg->model_value))
139  return -1;
140  if (rdcu_set_noise_bits_rounded(cfg->round))
141  return -1;
142 
143  if (rdcu_set_adaptive_1_golomb_param(cfg->ap1_golomb_par))
144  return -1;
145  if (rdcu_set_adaptive_1_spillover_threshold(cfg->ap1_spill))
146  return -1;
147 
148  if (rdcu_set_adaptive_2_golomb_param(cfg->ap2_golomb_par))
149  return -1;
150  if (rdcu_set_adaptive_2_spillover_threshold(cfg->ap2_spill))
151  return -1;
152 
153  if (rdcu_set_data_start_addr(cfg->rdcu_data_adr))
154  return -1;
155  if (rdcu_set_model_start_addr(cfg->rdcu_model_adr))
156  return -1;
157  if (rdcu_set_num_samples(cfg->samples))
158  return -1;
159  if (rdcu_set_new_model_start_addr(cfg->rdcu_new_model_adr))
160  return -1;
161 
162  if (rdcu_set_compr_data_buf_start_addr(cfg->rdcu_buffer_adr))
163  return -1;
164  if (rdcu_set_compr_data_buf_len(cfg->buffer_length))
165  return -1;
166 
167  /* now sync the configuration registers to the RDCU... */
168  if (rdcu_sync_compressor_param1())
169  return -1;
170  if (rdcu_sync_compressor_param2())
171  return -1;
172  if (rdcu_sync_adaptive_param1())
173  return -1;
174  if (rdcu_sync_adaptive_param2())
175  return -1;
176  if (rdcu_sync_data_start_addr())
177  return -1;
178  if (rdcu_sync_model_start_addr())
179  return -1;
180  if (rdcu_sync_num_samples())
181  return -1;
182  if (rdcu_sync_new_model_start_addr())
183  return -1;
184  if (rdcu_sync_compr_data_buf_start_addr())
185  return -1;
186  if (rdcu_sync_compr_data_buf_len())
187  return -1;
188 
189  /* wait for it */
190  rdcu_syncing();
191 
192  return 0;
193 }

229  * @brief set up RDCU SRAM for compression
230  *
231  * @param cfg pointer to a compression configuration
232  *
233  * @returns 0 on success, error otherwise
234  */
235 
236 static int rdcu_transfer_sram(const struct cmp_cfg *cfg)
237 {
238  if (cfg->input_buf != NULL) {
239  /* round up needed size must be a multiple of 4 bytes */
240  uint32_t size = (cfg->samples * 2 + 3) & ~3U;
241  /* now set the data in the local mirror... */
242  if (rdcu_write_sram_16(cfg->input_buf, cfg->rdcu_data_adr, cfg->samples * 2) < 0) {
243  debug_print("Error: The data to be compressed cannot be transferred to the SRAM of the RDCU.\n");
244  return -1;
245  }
246  if (rdcu_sync_mirror_to_sram(cfg->rdcu_data_adr, size, rdcu_get_data_mtu())) {
247  debug_print("Error: The data to be compressed cannot be transferred to the SRAM of the RDCU.\n");
248  return -1;
249  }
250  }
251  /*...and the model when needed */
252  if (cfg->model_buf != NULL) {
253  /* set model only when model mode is used */
254  if (model_mode_is_used(cfg->cmp_mode)) {
255  /* round up needed size must be a multiple of 4 bytes */
256  uint32_t size = (cfg->samples * 2 + 3) & ~3U;
257  /* set the model in the local mirror... */
258  if (rdcu_write_sram_16(cfg->model_buf, cfg->rdcu_model_adr, cfg->samples * 2) < 0) {
259  debug_print("Error: The model buffer cannot be transferred to the SRAM of the RDCU.\n");
260  return -1;
261  }
262  if (rdcu_sync_mirror_to_sram(cfg->rdcu_model_adr, size, rdcu_get_data_mtu())) {
263  debug_print("Error: The model buffer cannot be transferred to the SRAM of the RDCU.\n");
264  return -1;
265  }
266  }
267  }
268 
269  /* ...and wait for completion */
270  rdcu_syncing();
271 
272  return 0;
273 }

372  info->spill_used = rdcu_get_spillover_threshold();
373  info->model_value_used = (uint8_t)rdcu_get_weighting_param();
374  info->round_used = (uint8_t)rdcu_get_noise_bits_rounded();
375  info->rdcu_new_model_adr_used = rdcu_get_new_model_addr_used();
376  info->samples_used = rdcu_get_samples_used();

380  info->ap2_cmp_size = rdcu_get_compr_data_adaptive_2_size_bit();
381  info->cmp_err = rdcu_get_compr_error();

383  /* There is a bug up to RDCU FPGA version 0.7 where the
384  * compressed size is not updated accordingly in RAW mode.
385  */
386  if (info->cmp_mode_used == CMP_MODE_RAW) {
387  info->cmp_size = info->samples_used * IMA_SAM2BYT * 8;
388  info->ap1_cmp_size = info->cmp_size;
389  info->ap2_cmp_size = info->cmp_size;
390  }
391 #endif

395  * parameter is smaller than 3.
396  */
397  if (info->cmp_mode_used == CMP_MODE_RAW && info->samples_used < 3) {

400  info->ap2_cmp_size = info->cmp_size;
401  }
402 #endif
403  }
404  return 0;
405 }

497 /**
498  * @brief inject a SRAM edac multi bit error into the RDCU SRAM
499  *
500  * @param cfg configuration to inject error
501  * @param addr SRAM address to inject edac error
502  */
503 
504 int rdcu_inject_edac_error(const struct cmp_cfg *cfg, uint32_t addr)
505 {
506  uint32_t sub_chip_die_addr;
507  uint8_t buf[4] = {0};
508 
509  if (rdcu_set_compression_register(cfg))
510  return -1;
511 
512  if (rdcu_transfer_sram(cfg))
513  return -1;
514 
515  /* disable edac */
516  for (sub_chip_die_addr = 1; sub_chip_die_addr <= 4; sub_chip_die_addr++) {
517  rdcu_edac_set_sub_chip_die_addr(sub_chip_die_addr);
518  rdcu_edac_set_ctrl_reg_write_op();
519  rdcu_edac_set_bypass();
520  if (rdcu_sync_sram_edac_ctrl()) {
521  debug_print("Error: rdcu_sync_sram_edac_ctrl\n");
522  return -1;
523  }
524  rdcu_syncing();
525  /* verify bypass aktiv */
526  rdcu_edac_set_ctrl_reg_read_op();
527  if (rdcu_sync_sram_edac_ctrl()) {
528  debug_print("Error: rdcu_sync_sram_edac_ctrl\n");
529  return -1;
530  }
531  rdcu_syncing();
532  if (rdcu_sync_sram_edac_status()) {
533  printf("Error: rdcu_sync_sram_edac_status\n");
534  return -1;
535  }
536  rdcu_syncing();
537  if (rdcu_edac_get_sub_chip_die_addr() != sub_chip_die_addr) {
538  printf("Error: sub_chip_die_addr unexpected !\n");
539  return -1;
540  }
541 #if 1
542  /* It looks like there is a bug when displaying the bypass status of the 2. and 4. SRAM chip. */
543  if (2 != sub_chip_die_addr && 4 != sub_chip_die_addr)
544 #endif
545  if (rdcu_edac_get_bypass_status() == 0) {
546  printf("Error: bypass status unexpected !\n");
547  return -1;
548  }
549  }
550 
551  /* inject multi bit error */
552  if (rdcu_sync_sram_to_mirror(addr, sizeof(buf), rdcu_get_data_mtu()))
553  return -1;
554  rdcu_syncing();
555  if (rdcu_read_sram(buf, addr, sizeof(buf)) < 0)
556  return -1;
557 
558  buf[0] ^= 1 << 0;
559  buf[1] ^= 1 << 1;
560  buf[2] ^= 1 << 2;
561  buf[3] ^= 1 << 3;
562 
563  if (rdcu_write_sram(buf, addr, sizeof(buf)) < 0)
564  return -1;
565  if (rdcu_sync_mirror_to_sram(addr, sizeof(buf), rdcu_get_data_mtu())) {
566  debug_print("Error: The data to be compressed cannot be transferred to the SRAM of the RDCU.\n");
567  return -1;
568  }
569  rdcu_syncing();
570 
571 
572  /* enable edac again */
573  for (sub_chip_die_addr = 1; sub_chip_die_addr <= 4; sub_chip_die_addr++) {
574  if (rdcu_edac_set_sub_chip_die_addr(sub_chip_die_addr))
575  return -1;
576  rdcu_edac_set_ctrl_reg_write_op();
577  rdcu_edac_clear_bypass();
578  if (rdcu_sync_sram_edac_ctrl()) {
579  debug_print("Error: rdcu_sync_sram_edac_ctrl\n");
580  return -1;
581  }
582  rdcu_syncing();
583  /* verify bypass disable */
584  rdcu_edac_set_ctrl_reg_read_op();
585  if (rdcu_sync_sram_edac_ctrl()) {
586  debug_print("Error: rdcu_sync_sram_edac_ctrl\n");
587  return -1;
588  }
589  rdcu_syncing();
590  if (rdcu_sync_sram_edac_status()) {
591  printf("Error: rdcu_sync_sram_edac_status\n");
592  return -1;
593  }
594  rdcu_syncing();
595  if (rdcu_edac_get_sub_chip_die_addr() != sub_chip_die_addr) {
596  printf("Error: sub_chip_die_addr unexpected !\n");
597  return -1;
598  }
599  if (1 == rdcu_edac_get_bypass_status()) {
600  printf("Error: bypass status unexpected !\n");
601  return -1;
602  }
603  }
604  return 0;
605 }

608 /**
609  * @brief compressing data with the help of the RDCU hardware compressor; read
610  * data from the last compression before starting compression
611  *
612  * @param cfg configuration contains all parameters required for compression
613  * @param last_info compression information of last compression run
614  *
615  * @note when using the 1d-differencing mode or the raw mode (cmp_mode = 0,2,4),
616  * the model parameters (model_value, model_buf, rdcu_model_adr) are ignored
617  * @note the overlapping of the different rdcu buffers is not checked
618  * @note the validity of the cfg structure is checked before the compression is
619  * started
620  *
621  * @returns 0 on success, error otherwise
622  */
623 
624 int rdcu_compress_data_parallel(const struct cmp_cfg *cfg,
625  const struct cmp_info *last_info)
626 {
627  uint32_t samples_4byte;
628 
629  if (!cfg)
630  return -1;
631 
632  if (!last_info)
633  return rdcu_compress_data(cfg);
634 
635  if (last_info->cmp_err)
636  return -1;
637 
638  rdcu_set_compression_register(cfg);
639 
640  /* round up needed size must be a multiple of 4 bytes */
641  samples_4byte = (cfg->samples * IMA_SAM2BYT + 3) & ~3U;
642 
643  if (cfg->input_buf != NULL) {
644  uint32_t cmp_size_4byte;
645 
646  /* now set the data in the local mirror... */
647  if (rdcu_write_sram_16(cfg->input_buf, cfg->rdcu_data_adr,
648  cfg->samples * IMA_SAM2BYT) < 0)
649  return -1;
650 
651  /* calculate the need bytes for the bitstream */
652  cmp_size_4byte = cmp_bit_to_4byte(last_info->cmp_size);
653 
654  /* parallel read compressed data and write input data from sram
655  * to mirror */
656  if (rdcu_sync_sram_mirror_parallel(last_info->rdcu_cmp_adr_used,
657  cmp_size_4byte, cfg->rdcu_data_adr, samples_4byte,
658  rdcu_get_data_mtu()))
659  return -1;
660  /* wait for it */
661  rdcu_syncing();
662  if (cfg->icu_output_buf) {
663  if (rdcu_read_sram(cfg->icu_output_buf,

670  }

673  if (cfg->model_buf && model_mode_is_used(cfg->cmp_mode) && model_mode_is_used(last_info->cmp_mode_used)) {

679

681  if (rdcu_write_sram_16(cfg->model_buf, cfg->rdcu_model_adr,

688  cfg->rdcu_model_adr,
689  samples_4byte, rdcu_get_data_mtu()))
690  return -1;
691  /* wait for it */
692  rdcu_syncing();
693  if (cfg->icu_new_model_buf) {
694  if (rdcu_read_sram(cfg->icu_new_model_buf,
695  last_info->rdcu_new_model_adr_used,
696  new_model_size_4byte) < 0)
697  return -1;
698  }
699  }
700  /* write model */
701  } else if (cfg->model_buf && model_mode_is_used(cfg->cmp_mode)) {
702  /* set the model in the local mirror... */
703  if (rdcu_write_sram_16(cfg->model_buf, cfg->rdcu_model_adr,
704  cfg->samples * IMA_SAM2BYT) < 0)
705  return -1;
706 
707  if (rdcu_sync_mirror_to_sram(cfg->rdcu_model_adr, samples_4byte,
708  rdcu_get_data_mtu()))
709  return -1;
710  /* read model */
711  } else if (model_mode_is_used(last_info->cmp_mode_used)) {
712  if (rdcu_read_model(last_info, cfg->icu_new_model_buf) < 0)
713  return -1;
714  }
715 
716  /* ...and wait for completion */
717  rdcu_syncing();
718 
719  if (rdcu_start_compression())
720  return -1;
721 
722  return 0;
