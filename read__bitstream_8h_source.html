16  * @brief this library handles the reading from an MSB-first bitstream
17  *
18  * This API consists of small unitary functions, which must be inlined for best performance.
19  * Since link-time-optimization is not available for all compilers, these
20  * functions are defined into a .h to be included.
21  *
22  * Start by invoking bit_init_decoder(). A chunk of the bitstream is then stored
23  * into a local register. The local register size is 64 bits. You can then retrieve
24  * bit-fields stored into the local register. The local register is explicitly
25  * reloaded from the memory with the bit_refill() function.
26  * A reload guarantees a minimum of 57 bits in the local register if the
27  * returned status is BIT_UNFINISHED.
28  * Otherwise, it can be less than that, so proceed accordingly.
29  * Checking if bit_decoder has reached its end can be performed with bit_end_of_stream().
30  *
31  * This is based on the bitstream part of the FiniteStateEntropy library, see:
32  * https://github.com/Cyan4973/FiniteStateEntropy/blob/dev/lib/bitstream.h
33  * by @author Yann Collet
34  * As well as some ideas from this blog post:
35  * https://fgiesen.wordpress.com/2018/02/20/reading-bits-in-far-too-many-ways-part-2/
36  * by @author Fabian Giesen
37  */

113 /**
114  * @brief initialize a bit_decoder
115  *
116  * @param dec a pointer to an already allocated bit_decoder structure
117  * @param buf start address of the bitstream buffer
118  * @param buf_size size of the bitstream in bytes
119  *
120  * @returns size of stream (== src_size), or zero if a problem is detected
121  */
122 
123 static __inline size_t bit_init_decoder(struct bit_decoder *dec, const void *buf,
124  size_t buf_size)
125 {
126  if (buf_size < 1) {
127  memset(dec, 0, sizeof(*dec));
128  dec->bits_consumed = sizeof(dec->bit_container)*8;
129  return 0;
130  }
131 
132  dec->cursor = (const uint8_t *)buf;
133 
134  if (buf_size >= sizeof(dec->bit_container)) { /* normal case */
135  dec->bits_consumed = 0;
136  dec->bit_container = bit_read_unaligned_64be(dec->cursor);
137  dec->limit_ptr = dec->cursor + buf_size - sizeof(dec->bit_container);
138  } else {
139  const uint8_t *ui8_p = (const uint8_t *)(buf);
140 
141  dec->bits_consumed = (unsigned int)(sizeof(dec->bit_container) - buf_size) * 8;
142 
143  dec->bit_container = (uint64_t)ui8_p[0] << 56;
144  switch (buf_size) {
145  case 7:
146  dec->bit_container += (uint64_t)ui8_p[6] << 8;
147  /* fall-through */
148  case 6:
149  dec->bit_container += (uint64_t)ui8_p[5] << 16;
150  /* fall-through */
151  case 5:
152  dec->bit_container += (uint64_t)ui8_p[4] << 24;
153  /* fall-through */
154  case 4:
155  dec->bit_container += (uint64_t)ui8_p[3] << 32;
156  /* fall-through */
157  case 3:
158  dec->bit_container += (uint64_t)ui8_p[2] << 40;
159  /* fall-through */
160  case 2:
161  dec->bit_container += (uint64_t)ui8_p[1] << 48;
162  /* fall-through */
163  default:
164  break;
165  }
166  dec->bit_container >>= dec->bits_consumed;
167 
168  dec->limit_ptr = dec->cursor;
169  }
170  return buf_size;
171 }

174 /**
175  * @brief provides next n bits from local register; local register is not modified
176  *
177  * @param dec a pointer to a bit_decoder context
178  * @param nb_bits number of bits to look; only works if 1 <= nb_bits <= 56
179  *
180  * @returns extracted value
181  */
182 
183 static __inline uint64_t bit_peek_bits(const struct bit_decoder *dec, unsigned int nb_bits)
184 {
185  /* mask for the shift value register to prevent undefined behaviour */
186  uint32_t const reg_mask = 0x3F;
187 
188  assert(nb_bits >= 1 && nb_bits <= (64 - 7));
189  /* why -7: worst case refill can only put 56 bit in the bit_container */
190 
191  /* shift out consumed bits; return the top nb_bits bits we want to peek */
192  return (dec->bit_container << (dec->bits_consumed & reg_mask)) >> (64-nb_bits);
193 }

196 /**
197  * @brief count the leading ones in the local register; local register is not modified
198  *
199  * @param dec pointer to a bit_decoder context
200  *
201  * @returns number of leading ones;
202  */
203 
204 static __inline unsigned int bit_peek_leading_ones(const struct bit_decoder *dec)
205 {
206  /* mask for the shift value register to prevent undefined behaviour */
207  uint32_t const reg_mask = 0x3F;
208  /* shift out the bits we've already consumed */
209  uint64_t const remaining_flip = ~(dec->bit_container << (dec->bits_consumed & reg_mask));
210 
211  /* clzll(0) is undefined behaviour */
212  return remaining_flip ? (unsigned int)__builtin_clzll(remaining_flip) :
213  sizeof(dec->bit_container)*8;
214 }

230 /**
231  * @brief read and consume next n bits from the local register
232  * @warning do not read more bits than the local register has unconsumed bits.
233  * If you do this, the bit_refill function will return the BIT_OVERFLOW
234  * status the next time the register is refilled.
235  *
236  * @param dec pointer to a bit_decoder context
237  * @param nb_bits number of bits to look; only works if 1 <= nb_bits <= 56
238  *
239  * @returns extracted value
240  */
241 
242 static __inline uint64_t bit_read_bits(struct bit_decoder *dec, unsigned int nb_bits)
243 {
244  uint64_t const read_bits = bit_peek_bits(dec, nb_bits);
245 
246  bit_consume_bits(dec, nb_bits);
247  return read_bits;

271 /**
272  * @brief same as bit_read_bits32() but subtract 1 from the extracted value
273  * @warning do not read more bits than the local register has unconsumed bits.
274  * If you do this, the bit_refill function will return the BIT_OVERFLOW
275  * status the next time the register is refilled.
276  *
277  * @param dec pointer to a bit_decoder context
278  * @param nb_bits number of bits to read; only works if nb_bits <= 32
279  *
280  * @returns extracted 32 bit value minus 1
281  *
282  * @note The difference to the bit_read_bits32() function with subtraction is
283  * that the subtracted value is masked with nb_bits. E.g. if you read 4
284  * bits from the bitstream and get 0 and then subtract 1, you get 0xFF
285  * instead of 0xFFFFFFFF
286  */
287 
288 static __inline uint32_t bit_read_bits32_sub_1(struct bit_decoder *dec, unsigned int nb_bits)
289 {
290  /* mask for the shift value register to prevent undefined behaviour */
291  uint32_t const reg_mask = sizeof(dec->bit_container)*8 - 1;
292  unsigned int const shift_bits = (64 - dec->bits_consumed - nb_bits) & reg_mask;
293  uint32_t bits_unmask;
294 
295  assert(nb_bits <= 32);
296 
297  bits_unmask = (uint32_t)(dec->bit_container >> shift_bits);
298  bit_consume_bits(dec, nb_bits);
299  return (bits_unmask - 1) & BIT_MASK[nb_bits];
300 }

303 /**
304  * @brief refill the local register from the buffer previously set in
305  * bit_init_decoder()
306  *
307  * @param dec a bitstream decoding context
308  *
309  * @note this function is safe, it guarantees that it does not read beyond
310  * initialize buffer
311  *
312  * @returns the status of bit_decoder internal register;
313  * BIT_UNFINISHED: internal register is filled with at least _57-bits_
314  * BIT_END_OF_BUFFER: reached the end of the buffer, only some bits are left in the bitstream
315  * BIT_ALL_READ_IN: _all_ bits of the buffer have been consumed
316  * BIT_OVERFLOW: more bits have been consumed than contained in the local register
317  */
318 
319 static __inline int bit_refill(struct bit_decoder *dec)
320 {
321  unsigned int const bytes_consumed = dec->bits_consumed >> 3;
322 
323  if (dec->bits_consumed > sizeof(dec->bit_container)*8)
324  return BIT_OVERFLOW;
325 
326  if (dec->cursor + bytes_consumed < dec->limit_ptr) {
327  /* Advance the pointer by the number of full bytes we consumed */
328  dec->cursor += bytes_consumed;
329  /* Refill the bit container */
330  dec->bit_container = bit_read_unaligned_64be(dec->cursor);
331  /* The number of bits that we have already consumed in the
332  * current byte, excluding the bits that formed a complete byte
333  * and were already processed.
334  */
335  dec->bits_consumed &= 0x7;
336  return BIT_UNFINISHED;
337  }
338 
339  if (dec->cursor == dec->limit_ptr) {
340  if (dec->bits_consumed == sizeof(dec->bit_container)*8)
341  return BIT_ALL_READ_IN;
342  return BIT_END_OF_BUFFER;
343  }
344 
345  /* limit_ptr < (cursor + bytes_consumed) < end */
346  dec->bits_consumed -= (dec->limit_ptr - dec->cursor)*8;
347  dec->cursor = dec->limit_ptr;
348  dec->bit_container = bit_read_unaligned_64be(dec->cursor);
349 
350  return BIT_END_OF_BUFFER;
351 }
