70 __extension__
71 struct imagette_header {
72  uint16_t spill_used; /**< Spillover threshold used */
73  uint8_t golomb_par_used; /**< Golomb parameter used */
74  union{
75  struct {
76  uint8_t spare1;
77  uint8_t ima_cmp_dat[]; /**< compressed data for imagette specific header */
78  } __attribute__((packed));
79  struct {
80  uint16_t ap1_spill_used; /**< Adaptive Spillover threshold used 1 */
81  uint8_t ap1_golomb_par_used; /**< Adaptive Golomb parameter used 1 */
82  uint16_t ap2_spill_used; /**< Adaptive Spillover threshold used 2 */
83  uint8_t ap2_golomb_par_used; /**< Adaptive Golomb parameter used 2 */
84  uint8_t spare2;
85  uint16_t spare3;
86  uint8_t ap_ima_cmp_data[]; /**< compressed data for adaptive imagette specific header */
87  } __attribute__((packed));
88  };
89 } __attribute__((packed));
90 compile_time_assert(sizeof(struct imagette_header) == SPECIFIC_IMAGETTE_ADAPTIVE_HEADER_SIZE, AP_IMAGETTE_HEADER_T_SIZE_IS_NOT_CORRECT);

117 /**
118  * @brief definition of the compression entity format
119  */
120 
121 __extension__
122 struct cmp_entity {
123  uint32_t version_id; /**< ICU ASW/cmp_tool Version ID */
124  uint32_t cmp_ent_size:24; /**< Compression Entity Size */
125  uint32_t original_size:24; /**< Original Data Size */
126  union {
127  uint64_t start_timestamp:48; /**< Compression Start Timestamp */
128  struct timestamp_cmp_ent start_time;
129  } __attribute__((packed));
130  union {
131  uint64_t end_timestamp:48; /**< Compression End Timestamp */
132  struct timestamp_cmp_ent end_time;
133  } __attribute__((packed));
134  uint16_t data_type; /**< Data Product Type */
135  uint8_t cmp_mode_used; /**< used Compression Mode */
136  uint8_t model_value_used; /**< used Model Updating Weighing Value */
137  uint16_t model_id; /**< Model ID */
138  uint8_t model_counter; /**< Model Counter */
139  uint8_t max_used_bits_version;
140  uint16_t lossy_cmp_par_used; /**< used Lossy Compression Parameters */
141  union { /* specific Compression Entity Header for the different Data Product Types */
142  struct imagette_header ima;
143  struct non_imagette_header non_ima;
144  };
145 } __attribute__((packed));
146 compile_time_assert(sizeof(struct cmp_entity) == NON_IMAGETTE_HEADER_SIZE, CMP_ENTITY_SIZE_IS_NOT_CORRECT);
