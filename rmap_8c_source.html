412 /**
413  * @brief build an rmap header
414  *
415  * @param pkt a struct rmap_pkt
416  * @param hdr the header buffer; if NULL, the function returns the needed size
417  *
418  * @returns -1 on error, size of header otherwise
419  */
420 
421 int rmap_build_hdr(struct rmap_pkt *pkt, uint8_t *hdr)
422 {
423  int i;
424  int n = 0;
425 
426 
427  if (!pkt)
428  return -1;
429 
430  if (!hdr) {
431  n = rmap_get_min_hdr_size(pkt);
432  n += pkt->path_len;
433  n += pkt->rpath_len;
434  return n;
435  }
436 
437 
438  for (i = 0; i < pkt->path_len; i++)
439  hdr[n++] = pkt->path[i]; /* routing path to target */
440 
441  hdr[n++] = pkt->dst; /* target logical address */
442  hdr[n++] = pkt->proto_id; /* protocol id */
443  hdr[n++] = pkt->instruction; /* instruction */
444  hdr[n++] = pkt->key; /* key/status */
445 
446  for (i = 0; i < pkt->rpath_len; i++)
447  hdr[n++] = pkt->rpath[i]; /* return path to source */
448 
449  hdr[n++] = pkt->src; /* source logical address */
450  hdr[n++] = (uint8_t)(pkt->tr_id >> 8); /* MSB of transaction id */
451  hdr[n++] = (uint8_t)pkt->tr_id; /* LSB of transaction id */
452 
453 
454  /* commands have a data address */
455  if (pkt->ri.cmd_resp) {
456  hdr[n++] = 0x0; /* extended address field (unused) */
457  hdr[n++] = (uint8_t)(pkt->addr >> 24); /* data addr MSB */
458  hdr[n++] = (uint8_t)(pkt->addr >> 16);
459  hdr[n++] = (uint8_t)(pkt->addr >> 8);
460  hdr[n++] = (uint8_t)pkt->addr; /* data addr LSB */
461  } else if (!pkt->ri.cmd_resp && pkt->ri.cmd & RMAP_CMD_BIT_WRITE) {
462  /* all headers have data length unless they are a write reply */
463  return n;
464  } else {
465  hdr[n++] = 0x0; /* on other replies, this is a reserved field */
466  }
467 
468  hdr[n++] = (uint8_t)(pkt->data_len >> 16); /* data len MSB */
469  hdr[n++] = (uint8_t)(pkt->data_len >> 8);
470  hdr[n++] = (uint8_t)pkt->data_len; /* data len LSB */
471 
472  return n;
473 }

476 /**
477  * @brief create an rmap packet from a buffer
478  *
479  * @param buf the buffer, with the target path stripped away, i.e.
480  * starting with <logical address>, <protocol id>, ...
481  * @param len the data length of the buffer (in bytes)
482  *
483  * @returns an rmap packet, containing the decoded buffer including any data,
484  * NULL on error
485  */
486 
487 struct rmap_pkt *rmap_pkt_from_buffer(uint8_t *buf, uint32_t len)
488 {
489  size_t n = 0;
490  size_t i;
491  int min_hdr_size;
492 
493  struct rmap_pkt *pkt = NULL;
494 
495 
496  if (!buf)
497  goto error;
498 
499  if (len < RMAP_HDR_MIN_SIZE_WRITE_REP) {
500  printf("buffer len is smaller than the smallest RMAP packet\n");
501  goto error;
502  }
503 
504  if (buf[RMAP_PROTOCOL_ID] != RMAP_PROTOCOL_ID) {
505  printf("Not an RMAP packet, got %x but expected %x\n",
506  buf[RMAP_PROTOCOL_ID], RMAP_PROTOCOL_ID);
507  goto error;
508  }
509 
510  pkt = rmap_create_packet();
511  if (!pkt) {
512  printf("Error creating packet\n");
513  goto error;
514  }
515 
516  pkt->dst = buf[RMAP_DEST_ADDRESS];
517  pkt->proto_id = buf[RMAP_PROTOCOL_ID];
518  pkt->instruction = buf[RMAP_INSTRUCTION];
519  pkt->key = buf[RMAP_CMD_DESTKEY];
520 
521  min_hdr_size = rmap_get_min_hdr_size(pkt);
522  if (min_hdr_size < 0)
523  goto error;
524 
525  if (len < (uint32_t)min_hdr_size) {
526  printf("buffer len is smaller than the contained RMAP packet\n");
527  goto error;
528  }
529 
530 
531  if (pkt->ri.cmd_resp) {
532  pkt->rpath_len = (pkt->ri.reply_addr_len << 2) & 0xFF;
533  if (len < (uint32_t)min_hdr_size + pkt->rpath_len) {
534  printf("buffer is smaller than the contained RMAP packet\n");
535  goto error;
536  }
537 
538  pkt->rpath = (uint8_t *)malloc(pkt->rpath_len);
539  if (!pkt->rpath)
540  goto error;
541 
542  for (i = 0; i < pkt->rpath_len; i++)
543  pkt->rpath[i] = buf[RMAP_REPLY_ADDR_START + i];
544 
545  n = pkt->rpath_len; /* rpath skip */
546  }
547 
548  pkt->src = buf[RMAP_SRC_ADDR + n];
549  pkt->tr_id = (((uint16_t)buf[RMAP_TRANS_ID_BYTE0 + n] << 8) & 0xFF00) |
550  (uint16_t)buf[RMAP_TRANS_ID_BYTE1 + n];
551 
552  /* commands have a data address */
553  if (pkt->ri.cmd_resp) {
554  pkt->addr = ((uint32_t)buf[RMAP_ADDR_BYTE0 + n] << 24) |
555  ((uint32_t)buf[RMAP_ADDR_BYTE1 + n] << 16) |
556  ((uint32_t)buf[RMAP_ADDR_BYTE2 + n] << 8) |
557  (uint32_t)buf[RMAP_ADDR_BYTE3 + n];
558  n += 4; /* addr skip, extended byte is incorporated in define */
559  }
560 
561  /* all headers have data length unless they are a write reply */
562  if (!(!pkt->ri.cmd_resp && (pkt->ri.cmd & (RMAP_CMD_BIT_WRITE)))) {
563  pkt->data_len = ((uint32_t)buf[RMAP_DATALEN_BYTE0 + n] << 16) |
564  ((uint32_t)buf[RMAP_DATALEN_BYTE1 + n] << 8) |
565  (uint32_t)buf[RMAP_DATALEN_BYTE2 + n];
566  }
567 
568  pkt->hdr_crc = buf[RMAP_HEADER_CRC];
569 
570  if (pkt->data_len) {
571  size_t pkt_size = RMAP_DATA_START + n + pkt->data_len + 1; /* +1 for data CRC */
572 
573  if (len < pkt_size) {
574  printf("buffer len is smaller than the contained RMAP packet; buf len: %" PRIu32 " bytes vs RMAP: %zu bytes needed\n",
575  len, pkt_size);
576  goto error;
577  }
578  if (len > pkt_size)
579  printf("warning: the buffer is larger than the included RMAP packet\n");
580 
581  pkt->data = (uint8_t *)malloc(pkt->data_len);
582  if (!pkt->data)
583  goto error;
584 
585  for (i = 0; i < pkt->data_len; i++)
586  pkt->data[i] = buf[RMAP_DATA_START + n + i];
587 
588  /* final byte is data crc */
589  pkt->data_crc = buf[RMAP_DATA_START + n + i];
590  }
591 
592 
593  return pkt;
594 
595 error:
596  if (pkt) {
597  free(pkt->data);
598  free(pkt->rpath);
599  free(pkt);
600  }
601 
602  return NULL;
603 }

668 {

672 
673
