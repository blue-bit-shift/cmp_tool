323 /**
324  * @brief map a signed value into a positive value range
325  *
326  * @param value_to_map signed value to map
327  * @param max_data_bits how many bits are needed to represent the
328  * highest possible value
329  *
330  * @returns the positive mapped value
331  */
332 
333 static uint32_t map_to_pos(uint32_t value_to_map, unsigned int max_data_bits)
334 {
335  uint32_t result;
336  uint32_t mask = (~0U >> (32 - max_data_bits)); /* mask the used bits */
337 
338  value_to_map &= mask;
339  if (value_to_map >> (max_data_bits - 1)) { /* check the leading signed bit */
340  value_to_map |= ~mask; /* convert to 32-bit signed integer */
341  /* map negative values to uneven numbers */
342  result = (-value_to_map) * 2 - 1; /* possible integer overflow is intended */
343  } else {
344  /* map positive values to even numbers */
345  result = value_to_map * 2; /* possible integer overflow is intended */
346  }
347 
348  return result;
349 }

352 /**
353  * @brief put the value of up to 32 bits into a big endian bitstream
354  *
355  * @param value the value to put into the bitstream
356  * @param n_bits number of bits to put into the bitstream
357  * @param bit_offset bit index where the bits will be put, seen from
358  * the very beginning of the bitstream
359  * @param bitstream_adr this is the pointer to the beginning of the
360  * bitstream (can be NULL)
361  * @param max_stream_len maximum length of the bitstream in *bits*; is
362  * ignored if bitstream_adr is NULL
363  *
364  * @returns length in bits of the generated bitstream on success; returns
365  * negative in case of erroneous input; returns CMP_ERROR_SMALL_BUF if
366  * the bitstream buffer is too small to put the value in the bitstream
367  */
368 
369 static int put_n_bits32(uint32_t value, unsigned int n_bits, int bit_offset,
370  uint32_t *bitstream_adr, unsigned int max_stream_len)
371 {
372  /*
373  * UNSEGMENTED
374  * |-----------|XXXXXX|---------------|--------------------------------|
375  * |-bits_left-|n_bits|-------------------bits_right-------------------|
376  * ^
377  * local_adr
378  * SEGMENTED
379  * |-----------------------------|XXX|XXX|-----------------------------|
380  * |----------bits_left----------|n_bits-|---------bits_right----------|
381  */
382  unsigned int bits_left = bit_offset & 0x1F;
383  unsigned int bits_right = 64 - bits_left - n_bits;
384  unsigned int shift_left = 32 - n_bits;
385  int stream_len = (int)(n_bits + (unsigned int)bit_offset);
386  uint32_t *local_adr;
387  uint32_t mask;
388 
389  /* Leave in case of erroneous input */
390  if (bit_offset < 0)
391  return -1;
392 
393  if (n_bits == 0)
394  return stream_len;
395 
396  if ((int)shift_left < 0) /* check n_bits <= 32 */
397  return -1;
398 
399  if (!bitstream_adr) /* Do we need to write data to the bitstream? */
400  return stream_len;
401 
402  /* Check if bitstream buffer is large enough */
403  if ((unsigned int)stream_len > max_stream_len) {
404  debug_print("Error: The buffer for the compressed data is too small to hold the compressed data. Try a larger buffer_length parameter.\n");
405  return CMP_ERROR_SMALL_BUF;
406  }
407 
408  local_adr = bitstream_adr + (bit_offset >> 5);
409 
410  /* clear the destination with inverse mask */
411  mask = (0XFFFFFFFFU << shift_left) >> bits_left;
412  *(local_adr) &= ~mask;
413 
414  /* put (the first part of) the value into the bitstream */
415  *(local_adr) |= (value << shift_left) >> bits_left;
416 
417  /* Do we need to split the value over two words (SEGMENTED case) */
418  if (bits_right < 32) {
419  local_adr++; /* adjust address */
420 
421  /* clear the destination */
422  mask = 0XFFFFFFFFU << bits_right;
423  *(local_adr) &= ~mask;
424 
425  /* put the 2nd part of the value into the bitstream */
426  *(local_adr) |= value << bits_right;
427  }
428  return stream_len;
429 }

432 /**
433  * @brief forms the codeword according to the Rice code
434  *
435  * @param value value to be encoded (must be smaller or equal than cmp_ima_max_spill(m))
436  * @param m Golomb parameter, only m's which are power of 2 are allowed
437  * maximum allowed Golomb parameter is 0x80000000
438  * @param log2_m Rice parameter, is ilog_2(m) calculate outside function
439  * for better performance
440  * @param cw address where the code word is stored
441  *
442  * @warning no check of the validity of the input parameters!
443  * @returns the length of the formed code word in bits; code word is invalid if
444  * the return value is greater than 32
445  */
446 
447 static uint32_t rice_encoder(uint32_t value, uint32_t m, uint32_t log2_m,
448  uint32_t *cw)
449 {
450  uint32_t q = value >> log2_m; /* quotient of value/m */
451  uint32_t qc = (1U << q) - 1; /* quotient code without ending zero */
452 
453  uint32_t r = value & (m-1); /* remainder of value/m */
454  uint32_t rl = log2_m + 1; /* length of the remainder (+1 for the 0 in the quotient code) */
455 
456  *cw = (qc << (rl & 0x1FU)) | r; /* put the quotient and remainder code together */
457  /*
458  * NOTE: If log2_m = 31 -> rl = 32, (q << rl) leads to an undefined
459  * behavior. However, in this case, a valid code with a maximum of 32
460  * bits can only be formed if q = 0 and qc = 0. Any shift with 0 << x always
461  * results in 0, which forms the correct codeword in this case. To prevent
462  * undefined behavior, the right shift operand is masked (& 0x1FU)
463  */
464 
465  return rl + q; /* calculate the length of the code word */
466 }

469 /**
470  * @brief forms a codeword according to the Golomb code
471  *
472  * @param value value to be encoded (must be smaller or equal than cmp_ima_max_spill(m))
473  * @param m Golomb parameter (have to be bigger than 0)
474  * @param log2_m is ilog_2(m) calculate outside function for better performance
475  * @param cw address where the code word is stored
476  *
477  * @warning no check of the validity of the input parameters!
478  * @returns the length of the formed code word in bits; code word is invalid if
479  * the return value is greater than 32
480  */
481 
482 static uint32_t golomb_encoder(uint32_t value, uint32_t m, uint32_t log2_m,
483  uint32_t *cw)
484 {
485  uint32_t len = log2_m + 1; /* codeword length in group 0 */
486  uint32_t cutoff = (0x2U << log2_m) - m; /* members in group 0 */
487 
488  if (value < cutoff) { /* group 0 */
489  *cw = value;
490  } else { /* other groups */
491  uint32_t const reg_mask = 0x1FU; /* mask for the right shift operand to prevent undefined behavior */
492  uint32_t g = (value-cutoff) / m; /* group number of same cw length */
493  uint32_t r = (value-cutoff) - g * m; /* member in the group */
494  uint32_t gc = (1U << (g & reg_mask)) - 1; /* prepare the left side in unary */
495  uint32_t b = cutoff << 1; /* form the base codeword */
496 
497  *cw = gc << ((len+1) & reg_mask); /* composed codeword part 1 */
498  *cw += b + r; /* composed codeword part 2 */
499  len += 1 + g; /* length of the codeword */
500  }
501  return len;
502 }

541  * success; negative on error, CMP_ERROR_SMALL_BUF if the bitstream buffer

543  *
544  * @note no check if the data or model are in the allowed range
545  * @note no check if the setup->spillover_par is in the allowed range
546  */
547 
548 static int encode_value_zero(uint32_t data, uint32_t model, int stream_len,
549  const struct encoder_setupt *setup)
550 {
551  data -= model; /* possible underflow is intended */
552 
553  data = map_to_pos(data, setup->max_data_bits);
554 
555  /* For performance reasons, we check to see if there is an outlier
556  * before adding one, rather than the other way around:
557  * data++;
558  * if (data < setup->spillover_par && data != 0)
559  * return ...
560  */
561  if (data < (setup->spillover_par - 1)) { /* detect non-outlier */
562  data++; /* add 1 to every value so we can use 0 as escape symbol */
563  return encode_normal(data, stream_len, setup);
564  }
565 
566  data++; /* add 1 to every value so we can use 0 as escape symbol */
567 
568  /* use zero as escape symbol */
569  stream_len = encode_normal(0, stream_len, setup);
570  if (stream_len <= 0)
571  return stream_len;
572 
573  /* put the data unencoded in the bitstream */
574  stream_len = put_n_bits32(data, setup->max_data_bits, stream_len,
575  setup->bitstream_adr, setup->max_stream_len);
576

598 static int encode_value_multi(uint32_t data, uint32_t model, int stream_len,
599  const struct encoder_setupt *setup)
600 {
601  uint32_t unencoded_data;
602  unsigned int unencoded_data_len;
603  uint32_t escape_sym, escape_sym_offset;
604 
605  data -= model; /* possible underflow is intended */
606 
607  data = map_to_pos(data, setup->max_data_bits);
608 
609  if (data < setup->spillover_par) /* detect non-outlier */
610  return encode_normal(data, stream_len, setup);
611 
612  /*
613  * In this mode we put the difference between the data and the spillover
614  * threshold value (unencoded_data) after an encoded escape symbol, which
615  * indicate that the next codeword is unencoded.
616  * We use different escape symbol depended on the size the needed bit of
617  * unencoded data:
618  * 0, 1, 2 bits needed for unencoded data -> escape symbol is spillover_par + 0
619  * 3, 4 bits needed for unencoded data -> escape symbol is spillover_par + 1
620  * 5, 6 bits needed for unencoded data -> escape symbol is spillover_par + 2
621  * and so on
622  */
623  unencoded_data = data - setup->spillover_par;
624 
625  if (!unencoded_data) /* catch __builtin_clz(0) because the result is undefined.*/
626  escape_sym_offset = 0;
627  else
628  escape_sym_offset = (31U - (uint32_t)__builtin_clz(unencoded_data)) >> 1;
629 
630  escape_sym = setup->spillover_par + escape_sym_offset;
631  unencoded_data_len = (escape_sym_offset + 1U) << 1;
632 
633  /* put the escape symbol in the bitstream */
634  stream_len = encode_normal(escape_sym, stream_len, setup);
635  if (stream_len <= 0)
636  return stream_len;
637 
638  /* put the unencoded data in the bitstream */
639  stream_len = put_n_bits32(unencoded_data, unencoded_data_len, stream_len,
640  setup->bitstream_adr, setup->max_stream_len);
641 
642  return stream_len;
643 }

674  * the bitstream
675  *
676  * @param data data to encode
677  * @param model model of the data (0 if not used)
678  * @param stream_len length of the bitstream in bits
679  * @param setup pointer to the encoder setup
680  *
681  * @returns the bit length of the bitstream with the added encoded value on
682  * success; negative on error, CMP_ERROR_SMALL_BUF if the bitstream buffer
683  * is too small to put the value in the bitstream, CMP_ERROR_HIGH_VALUE if
